\begin{multicols}{2}
\begin{flushleft}
\begin{verse}[\versewidth]
%\raggedyright
\initialV{D}{ies} irae, dies illa\\*
Solvet saeclum in favílla,   \\*   
Teste David cum Sibýlla.        
                                
\flagverse{2}Quantus tremor est futúrus,    \\* 
Quando judex est ventúrus       \\*
Cuncta stricte discussúrus!     

\flagverse{3}Tuba mirum spargens sonum       \\*
Per sepúlcra regiónum,          \\*
Coget omnes ante thronum.       
                                
\flagverse{4}Mors stupébit et natúra,        \\*
Cum resúrget creatúra           \\*
Judicánti responsúra.           
                                
\flagverse{5}Liber scriptus proferétur,      \\*
In quo totum continétur         \\*
Unde mundus judicétur.          
                                
                                
\flagverse{6}Judex ergo cum sedébit,         \\*
Quidquid latet apparébit:       \\*
Nil inúltum remanébit.          
                                
                                
                                
                                
                                
                                
\flagverse{7}Quid sum miser tunc\\*
\ \qquad dictúrus?   \\*
Quem patrónum rogatúrus?        \\*
Cum vix justus sit secúrus.     
                                
\flagverse{8}Rex treméndae majestátis,       \\*
Qui salvándos salvas gratis,    \\*
Salva me, fons pietátis.        
                                
\flagverse{9}Recordáre, Jesu pie,            \\*
Quod sum causa tuae viae:       \\*
Ne me perdas illa die.          
                                
\flagverse{10}Quaerens me sedísti lassus:     \\*
Redemísti crucem passus:        \\*
Tantus labor non sit cassus.    
                                
                                
\flagverse{11}Juste judex ultiónis,           \\*
Donum fac remissiónis,          \\*
Ante diem ratiónis.             
                                
\flagverse{12}Ingemísco, tamquam reus:        \\*
Culpa rubet vultus meus:        \\*
Supplicánti parce Deus.         
                                
                                
                                
                                
                                
                                
\flagverse{13}Qui Maríam absolvísti     \\*
Et latrónem exaudísti,          \\*
Mihi quoque spem dedísti.       
                                
\flagverse{14}Preces meae non sunt dignae:    \\*
Sed tu, bonus, fac benígne,     \\*
Ne perénni cremer igne.         
                                
\flagverse{15}Inter oves locum praesta,       \\*
Et ab haedis me sequéstra,      \\*
Statúens in parte dextra.       
                                
\flagverse{16}Confutátis maledíctis,          \\*
Flammis ácribus addíctis:       \\*
Voca me cum benedíctis.         
                                
\flagverse{17}Oro supplex et acclínis,        \\*
Cor contrítum quasi cinis:      \\*
Gere curam mei finis.           
                                
\flagverse{18}Lacrimósa dies illa,            \\*
Qua resúrget ex favílla\\*
Judicándus homo reus:           \\*
Huic ergo parce, Deus.          
                                
\flagverse{19}Pie Jesu Dómine,                \\*
dona eis réquiem.               \\*
Amen.                           

\end{verse}
\end{flushleft}
\end{multicols}
