\newHymn
\FirstLine{Day of wrath, O day of mourning!}

\begin{verse}[\versewidth]

\FirstVerse{D}{ay} of wrath, O day of mourning!\\*
        See fulfilled the Prophet's warning,\\*
        Heaven and earth in ashes burning.
	\pointorig

\flagverse{2} Oh, what fear man's bosom rendeth\\*
        When from heav'n the Judge descendeth\\*
        On whose sentence all dependeth!

\flagverse{3} Wondrous sound the trumpet flingeth,\\*
        Thro' earth's sepulchers it ringeth,\\*
        All before the throne it bringeth.

\flagverse{4} Death is struck and nature quaking;\\*
        All creation is awaking,\\*
        To its Judge an answer making.



\flagverse{5} Lo, the book, exactly worded,\\*
        Wherein all hath been recorded;\\*
        Thence shall judgment be awarded.

\flagverse{6} When the Judge His seat attaineth\\*
        And each hidden deed arraigneth,\\*
        Nothing unavenged remaineth.



\flagverse{7} What shall I, frail man, be pleading?\\*
        Who for me be interceding\\*
        When the just are mercy needing?

\flagverse{8} King of majesty tremendous,\\*
        Who dost free salvation send us,\\*
        Fount of pity, then befriend us.

\flagverse{9} Think, good Jesus, my salvation\\*
        Caused Thy wondrous incarnation;\\*
        Leave me not to reprobation!

\flagverse{10} Faint and weary Thou hast sought me,\\*
        On the cross of suffering bought me;\\*
        Shall such grace be vainly brought me?

\flagverse{11} Righteous Judge, for sin's pollution\\*
        Grant Thy gift of absolution\\*
        Ere that day of retribution!

\flagverse{12} Guilty, now I pour my moaning,\\*
        All my shame with anguish owning:\\*
        Spare, O God, Thy suppliant groaning!

\flagverse{13} Thou that sinful woman savedst,\\*
        Thou the dying thief forgavest,\\*
        And to me a hope vouchsafest.



\flagverse{14} Worthless are my prayers and sighing;\\*
        Yet, good Lord, in grace complying,\\*
        Rescue me from fires undying.

\flagverse{15} With Thy favoured sheep, oh, place me!\\*
        Nor among the goats abase me,\\*
        But to Thy right hand upraise me.

\flagverse{16} While the wicked are confounded,\\*
        Doomed to flames of woe unbounded,\\*
        Call me, with Thy saints surrounded.

\flagverse{17} Low I kneel with heart-submission,\\*
        See, like ashes, my contrition;\\*
        Help me in my last condition!

\flagverse{18} Ah! that day of tears and mourning!\\*
From the dust of earth returning,\\*
Man for judgement must prepare him!

\flagverse{19} Spare, O God, in mercy spare him!\\*
Lord, who didst our souls redeem,\\*
Grant a blessed requiem.

\end{verse}

\Htrans{William Josiah Irons}{1812--83}
