\newHymn

\FirstLine{Dear Angel"! ever at my side}

\begin{verse}[\versewidth]
\begin{altverse}
\FirstVerse{D}{ear} Angel! ever at my side,\\*
How loving must thou be,\\
To leave thy home in Heaven to guard\\*
A guilty wretch like me.
\end{altverse}

\begin{altverse}
\flagverse{2}Thy beautiful and shining face\\*
I see not, though so near;\\
The sweetness of thy soft low voice\\*
I am too deaf to hear.
\end{altverse}

\begin{altverse}
\flagverse{3}But when, dear Spirit! I kneel down\\*
Morning and night to prayer,\\
Something there is within my heart\\*
Which tells me thou art there.
\end{altverse}

\begin{altverse}
\flagverse{4}Yes! when I pray thou prayest too,\\*
Thy prayer is all for me;\\
But when I sleep thou sleepest not,\\*
But watchest patiently.
\end{altverse}

\begin{altverse}
\flagverse{5}Ah me! how lovely they must be\\*
Whom God has glorified;\\
Yet one of them, O sweetest thought!\\*
Is ever at my side.
\end{altverse}

\begin{altverse}
\flagverse{6}Then love me, love me, Angel dear!\\*
And I will love thee more;\\
And help me when my soul is cast\\*
Upon th'eternal shore.
\end{altverse}

\end{verse}

\Hpoet{Frederick William Faber}{1814--63}
\Hsource{Arundel Hymns}{1901}
