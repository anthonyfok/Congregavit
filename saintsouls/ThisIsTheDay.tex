\newHymn

\FirstLine{This is the day whereon the Lord's true witness}

\begin{verse}[\versewidth]
\indentpattern{0001}
\begin{patverse}
\FirstVerse{T}{his} is the day whereon the Lord's true witness,\\*
Whom all the nations lovingly do honour\\
Worthy at last was found to wear forever\\*
Glory transcendent.
\end{patverse}

\begin{patverse}
\flagverse{2} Loving, far-seeing, lowly, modest minded,\\*
So kept he well an even course unstained,\\
Ever while in his frame of manhood lingered\\*
Life's fitful breathings.
\end{patverse}

\begin{patverse}
\flagverse{3} Oft hath it been thro' his sublime deserving\\*
Poor human bodies, howsoever stricken,\\
Broke and cast off the bondage of their sickness,\\*
Healed Divinely.
\end{patverse}

\begin{patverse}
\flagverse{4} Wherefore to him we raise the solemn chorus,\\*
Chanting his praise and his surpassing triumph;\\
So may his pleading help us in the battle\\*
All through the ages.
\end{patverse}

\begin{patverse}
\flagverse{5} Healing and power, grace and beauteous honour\\*
Always be His, who shining in the highest,\\
Ruleth and keepeth all the world's vast order,\\*
One God three Persons.
\end{patverse}

\end{verse}

\Htrans{John O'Connor}{1870--1952}
