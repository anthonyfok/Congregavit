\newHymn
\FirstLine{Let all mortal flesh keep silence}

\begin{verse}[\versewidth]
\begin{altverse}
\FirstVerse{L}{et} all mortal flesh keep silence,\\*
And with fear and trembling stand;\\*
Ponder nothing earthly-minded,\\*
For with blessing in His hand,\\
Christ our God to earth descendeth,\\*
Our full homage to demand. 
\end{altverse}

\begin{altverse}
\flagverse{2} King of kings, yet born of Mary,\\*
As of old on earth He stood,\\
Lord of lords in human vesture,\\*
In the Body and the Blood\\
He will give to all the faithful\\*
His own self for heavenly
food. 
\end{altverse}

\begin{altverse}
\flagverse{3} Rank on rank the host of heaven\\*
Spreads its vanguard on the way.\\
As the Light of Light descendeth\\*
From the realms of endless day,\\
That the powers of hell may vanish\\*
As the darkness clears away.
\end{altverse}

\begin{altverse}
\flagverse{4} At His feet the six-winged Seraph,\\*
Cherubim with sleepless eye,\\
Veil their faces to the presence,\\*
As with ceaseless voice they cry,\\
Alleluia, Alleluia,\\*
Alleluia, Lord most high.
\end{altverse}

\end{verse}
