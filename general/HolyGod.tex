\newHymn
\FirstLine{Holy God, we praise Thy name}

\begin{verse}[\versewidth]
\indentpattern{010100}
\begin{patverse}
\FirstVerse{H}{oly} God, we praise Thy name;\\*
Lord of all, we bow before Thee\\*
All on earth Thy sceptre claim,\\*
All in heaven above adore Thee.\pointorig\\*
Infinite Thy vast domain,\\*
Everlasting is Thy reign.
\end{patverse}

\begin{patverse}
\flagverse{2}Hark! the loud celestial hymn,\\*
Angel choirs above are raising;\\
Cherubim and seraphim,\\*
In unceasing chorus praising,\\
Fill the heavens with sweet accord,\\*
Holy, holy, holy Lord.
\end{patverse}

\begin{patverse}
\flagverse{3}Lo, the Apostolic train\\*
 Join, Thy sacred name to hallow:\\
 Prophets swell the loud refrain,\\*
 And the white-robed Martyrs follow;\\
 And, from morn till set of sun,\\*
 Through the Church the song goes on.
\end{patverse}


\begin{patverse}
\flagverse{4}Holy Father, holy Son,\\*
Holy Spirit, Three we name Thee,\\
While in Essence only One\\*
Undivided God we 'claim Thee;\\
And adoring bend the knee,\\*
While we own the mystery.
\end{patverse}

\begin{patverse}
\flagverse{5}Thou art King of glory, Christ:\\*
Son of God, yet born of Mary;\\     
For us sinners sacrificed,\\*
And to death a tributary:\\
First to break the bars of death,\\*
Thou has opened heaven to faith.                
\end{patverse}

\begin{patverse}
\flagverse{6}From Thy high celestial home,\\*
Judge of all, again returning,\\
We believe that Thou shalt come\\*
In the dreaded Doomsday morning;\\
When Thy voice shall shake the earth,\\*
And the startled dead come forth.           
\end{patverse}

\begin{patverse}
\flagverse{7}Spare Thy people, Lord, we pray,\\*
By a thousand snares surrounded:\\
Keep us without sin today, \\*  
Never let us be confounded.\\
Lo, I put my trust in Thee;\\*
Never, Lord, abandon me. 
\end{patverse}

\end{verse}
