\newHymn

\FirstLine{Firmly I believe and truly}

\begin{verse}[\versewidth]
\begin{altverse}

\FirstVerse{F}{irmly} I believe and
truly\\*
God is Three and God is One\\*
And I next acknowledge duly\\*
Manhood taken by the Son;
\end{altverse}

\begin{altverse}
\flagverse{2} And I trust and hope most fully\\*
In that Manhood crucified;\\*
And each thought and deed unruly\\*
Do to death, as He has died.
\end{altverse}

\begin{altverse}
\flagverse{3}Simply to His grace and wholly\\*
Light and life and strength belong;\\*
And I love supremely, solely,\\*
Him the holy, Him the strong.
\end{altverse}

\begin{altverse}
\flagverse{4}And I hold in veneration,\\*
For the love of Him alone,\\*
Holy Church, as His creation,\\*
And her teachings, as His own.
\end{altverse}

\begin{altverse}
\flagverse{5}Adoration aye be given,\\*
With and through the angelic host,\\*
To the God of earth and heaven,\\*
Father, Son, and Holy Ghost.

\end{altverse}
\end{verse}

\Hpoet{Bl.~John Henry Cardinal Newman}{1801--90}
