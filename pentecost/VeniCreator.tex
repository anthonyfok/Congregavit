\newHymn
\FirstLine{Veni, Creator Spiritus}

\begin{verse}[\versewidth]

\FirstVerse{V}{eni,} Creator Spiritus,\\*
 mentes tuorum visita,        \\*
 imple superna gratia         \\*
 quae tu creasti pectora.     

\flagverse{2} Qui diceris Paraclitus,\\*
 altissimi donum Dei,     \\*    
 fons vivus, ignis, caritas,\\*  
 et spiritalis unctio.        

\flagverse{3} Tu, septiformis munere,\\*
 digitus paternae dexterae,\\*
 Tu rite promissum Patris,   \\* 
 sermone ditans guttura.      

\flagverse{4} Accende lumen sensibus:\\*
 infunde amorem cordibus: \\*    
 infirma nostri corporis    \\*  
 virtute firmans perpeti.     

\flagverse{5} Hostem repellas longius,\\*
 pacemque dones protinus:  \\*   
 ductore sic te praevio      \\* 
 vitemus omne noxium.         

\flagverse{6} Per te sciamus da Patrem,\\*
 noscamus atque Filium;     \\*  
 Teque utriusque Spiritum     \\*
 credamus omni tempore.       

\flagverse{7} Deo Patri sit gloria,\\*
 et Filio, qui a mortuis\\*      
 surrexit, ac Paraclito,  \\*    
 in saeculorum saecula.       
 Amen.                        

\end{verse}

\Hpoet{Rabanus Maurus}{776--856}
